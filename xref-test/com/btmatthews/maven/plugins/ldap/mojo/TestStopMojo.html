36  /**
37   * Unit tests for the Mojo that implements the stop goal.
38   *
39   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
40   * @since 1.1.0
41   */
42  public class TestStopMojo {
43  
44      /**
45       * Start a mock server and verify that the {@link StopLDAPMojo} signals it to shutdown.
46       *
47       * @throws Exception If the test case failed.
48       */
49      @Test
50      public void testStop() throws Exception {
51          final int port = TestUtils.getUnusedPort(10389);
52          final Server server = mock(Server.class);
53          when(server.isStarted(any(Logger.class))).thenReturn(true);
54          final MonitorObserver observer = mock(MonitorObserver.class);
55          final Logger logger = mock(Logger.class);
56          final Monitor monitor = new Monitor("ldap", port);
57          final Thread monitorThread = monitor.runMonitorDaemon(server, logger, observer);
58          final Timer timer = new Timer();
59          timer.schedule(new StopLDAPTask(timer, port), 5000L);
60          monitorThread.join(15000L);
61          verify(server).start(same(logger));
62          verify(logger).logInfo(eq("Waiting for command from client"));
63          verify(logger).logInfo(eq("Receiving command from client"));
64          verify(server).stop(same(logger));
65          validateMockitoUsage();
66      }
67  
68      public static class StopLDAPTask extends TimerTask {
69  
70          private final Timer timer;
71          private final int port;
72  
73          public StopLDAPTask(final Timer timer, final int port) {
74              this.timer = timer;
75              this.port = port;
76          }
77  
78          @Override
79          public void run() {
80              try {
81                  System.out.println("Sending stop");
82                  final StopLDAPMojo mojo = new StopLDAPMojo();
83                  setVariableValueInObject(mojo, "monitorPort", port);
84                  setVariableValueInObject(mojo, "monitorKey", "ldap");
85                  mojo.execute();
86              } catch (final Exception e) {
87                  timer.schedule(new StopLDAPTask(timer, port), 100L);
88              }
89          }
90      }
