33  /**
34   * Unit tests for the dump plugin goal.
35   *
36   * @author <a href="mailto:brian.matthews@btmatthews.com">Brian Matthews</a>
37   * @since 1.2.0
38   */
39  @DirectoryServerConfiguration(ldifFile = "com/btmatthews/maven/plugins/ldap/mojo/initial.ldif")
40  public final class TestDumpMojo {
41  
42      /**
43       * Temporary folder in which the mojo will dump data.
44       */
45      @Rule
46      public final TemporaryFolder outputDirectory = new TemporaryFolder();
47  
48      @Rule
49      public final DirectoryServerRule directoryServerRule = new DirectoryServerRule();
50  
51      /**
52       * The mojo that implements the dump goal.
53       */
54      public final DumpMojo mojo = new DumpMojo();
55  
56      /**
57       * Prepare for test case execution by creating and initialising the test case fixtures and mock objects.
58       *
59       * @throws Exception If there was a problem configuring the mojo.
60       */
61      @Before
62      public void setUp() throws Exception {
63          mojo.setLog(new SystemStreamLog());
64          setVariableValueInObject(mojo, "outputDirectory", outputDirectory.getRoot());
65          setVariableValueInObject(mojo, "host", "localhost");
66          setVariableValueInObject(mojo, "port", 10389);
67          setVariableValueInObject(mojo, "authDn", "uid=admin,ou=system");
68          setVariableValueInObject(mojo, "passwd", "secret");
69          setVariableValueInObject(mojo, "searchBase", "dc=btmatthews,dc=com");
70          setVariableValueInObject(mojo, "searchFilter", "(objectclass=*)");
71      }
72  
73      /**
74       * Test the configuration for the dump goal.
75       *
76       * @throws Exception If something unexpected happens.
77       */
78      @Test
79      public void dumpDSML() throws Exception {
80          setVariableValueInObject(mojo, "filename", "dump.dsml");
81          setVariableValueInObject(mojo, "format", "dsml");
82          mojo.execute();
83          assertTrue(new File(outputDirectory.getRoot(), "dump.dsml").exists());
84      }
85  
86      /**

90       */
91      @Test
92      public void dumpLDIF() throws Exception {

94          setVariableValueInObject(mojo, "format", "ldif");
95          mojo.execute();
96          assertTrue(new File(outputDirectory.getRoot(), "dump.ldif").exists());
97      }
