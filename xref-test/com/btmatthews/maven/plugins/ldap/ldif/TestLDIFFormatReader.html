35  /**
36   * Unit test the {@link LDIFFormatReader}.
37   *
38   * @author <a href="mailto:brian@btmatthews.com">Brian Matthews</a>
39   * @since 1.2.0
40   */
41  public class TestLDIFFormatReader {
42  
43      /**
44       * Verify that the {@link IOException} thrown by the {@link InputStream} is propagated.
45       *
46       * @throws Exception Expecting an {@link IOException} all other exceptions are unexpected.
47       */
48      @Test(expected = IOException.class)
49      public void handlesIOException() throws Exception {
50          final InputStream inputStream = mock(InputStream.class);
51          when(inputStream.available()).thenReturn(1000);
52          doThrow(IOException.class).when(inputStream).read(any(byte[].class), anyInt(), anyInt());
53          final FormatReader reader = new LDIFFormatReader(inputStream);
54          reader.nextRecord();
55      }
56  
57      /**
58       * Verify the {@link LDIFFormatReader} behaves correctly when loading an empty LDIF file.
59       *
60       * @throws Exception If there was an error in the test case.
61       */
62      @Test
63      public void canReadEmptyFile() throws Exception {
64          final InputStream inputStream = TestLDIFFormatReader.class.getResourceAsStream("empty.ldif");
65          assertNotNull(inputStream);
66          final FormatReader reader = new LDIFFormatReader(inputStream);
67          final LDIFChangeRecord first = reader.nextRecord();
68          assertNull(first);
69          reader.close();
70          inputStream.close();
71      }
72  
73      /**
74       * Verify the {@link LDIFFormatReader} behaves correctly when loading a LDIF file with a single
75       * change record.
76       *
77       * @throws Exception If there was an error in the test case.
78       */
79      @Test
80      public void canReadFileWithOneItem() throws Exception {
81          final InputStream inputStream = TestLDIFFormatReader.class.getResourceAsStream("one.ldif");
82          assertNotNull(inputStream);
83          final FormatReader reader = new LDIFFormatReader(inputStream);
84          final LDIFChangeRecord first = reader.nextRecord();
85          assertNotNull(first);
86          assertEquals("ou=People,dc=btmatthews,dc=com", first.getDN());
87          assertEquals(ChangeType.ADD, first.getChangeType());
88          final LDIFChangeRecord second = reader.nextRecord();
89          Assert.assertNull(second);
90          reader.close();
91          inputStream.close();
92      }
93  
94      /**
95       * Verify the {@link LDIFFormatReader} behaves correctly when loading a LDIF file with two
96       * change records.
97       *
98       * @throws Exception If there was an error in the test case.
99       */
100     @Test
101     public void canReadFileWithTwoItems() throws Exception {
102         final InputStream inputStream = TestLDIFFormatReader.class.getResourceAsStream("two.ldif");
103         assertNotNull(inputStream);
104         final FormatReader reader = new LDIFFormatReader(inputStream);
105         final LDIFChangeRecord first = reader.nextRecord();
106         assertNotNull(first);
107         assertEquals("ou=People,dc=btmatthews,dc=com", first.getDN());
108         assertEquals(ChangeType.ADD, first.getChangeType());
109         final LDIFChangeRecord second = reader.nextRecord();
110         assertNotNull(second);
111         assertEquals("cn=Bart Simpson,ou=People,dc=btmatthews,dc=com", second.getDN());
112         assertEquals(ChangeType.ADD, second.getChangeType());
113         final LDIFChangeRecord third = reader.nextRecord();
114         Assert.assertNull(third);
115         reader.close();
116         inputStream.close();
117     }
